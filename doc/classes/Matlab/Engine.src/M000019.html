    def method_missing(method_id, *args)
      method_name = method_id.id2name
      
      if method_name[(-1)..-1] == "="
        @driver.put_variable(@handle, method_name.chop, args.first)
      else
        @driver.get_variable(@handle, method_name)
      end
    end
